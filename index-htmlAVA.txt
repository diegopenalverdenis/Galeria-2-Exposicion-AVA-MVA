<script>
  // Volumen inicial
  document.getElementById("ambientAudio").volume = 0.5;

  // Toggle de audio + info curatorial
  function toggleAudio() {
    var audio = document.getElementById("ambientAudio");
    var icon = document.getElementById("audioIcon");
    var info = document.getElementById("infoCuratorial");

    if (audio.paused) {
      audio.play();
      icon.src = "imagenes/sound-on-icon-icon-md.png";
      info.classList.add("visible");
    } else {
      audio.pause();
      icon.src = "imagenes/sound-off-icon-md.png";
      info.classList.remove("visible");
    }
  }

  // Mostrar info curatorial al cargar (fade-in inicial)
  window.addEventListener("load", function() {
    document.getElementById("infoCuratorial").classList.add("visible");
  });
</script>

<script>
  // === BLOQUE CURATORIAL: Visor de imagen funcional y reversible ===
  // Fecha: 2025-11-20
  // Función: Abre visor de imagen, limpia paneles flotantes, evita reactivación

  window.visorActivo = false; // Bandera global

  function openPopup(imageSrc, title, details) {
    if (!imageSrc || imageSrc.trim() === "") return;

    window.visorActivo = true;

    // Limpieza visual al abrir visor
    document.querySelectorAll('.info-hotspot-text').forEach(el => {
      el.style.display = 'none';
    });
    document.querySelectorAll('.info-hotspot-title').forEach(el => {
      el.style.display = 'none';
    });
    document.querySelectorAll('.info-hotspot-header').forEach(el => {
      el.classList.add('bloqueado');
    });

    // Carga de imagen y contenido
    const img = document.getElementById("popupImage");
    img.style.display = "none";
    img.src = imageSrc;
document.getElementById("popupZoom").href = imageSrc;

    img.onload = () => {
      img.style.display = "block";
    };

img.onclick = () => {
  img.classList.toggle("zoomed");
};

    document.getElementById("popupTitle").innerHTML = title;
document.getElementById("popupTitleHeader").innerHTML = title;
    document.getElementById("popupDetails").innerHTML = details;
// Crear enlace para abrir imagen en nueva pestaña

    document.getElementById("popupOverlay").style.display = "flex";
  }

  // Cierre del visor
  document.getElementById("popupOverlay").addEventListener("click", function(e) {
    if (e.target.id === "popupOverlay") {
      this.style.display = "none";
      window.visorActivo = false;

      document.querySelectorAll('.info-hotspot-title').forEach(el => {
        el.style.removeProperty('display');
      });
      document.querySelectorAll('.info-hotspot-header').forEach(el => {
        el.classList.remove('bloqueado');
      });
    }
  });
</script>
